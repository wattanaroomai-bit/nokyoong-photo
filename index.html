/** * 1. ฟังก์ชันหลักสำหรับเปิดหน้าเว็บ HTML */ function doGet(e) { return HtmlService.createHtmlOutputFromFile('index') .setTitle('NOKYOONG PHOTO') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL) .addMetaTag('viewport', 'width=device-width, initial-scale=1, maximum-scale=1'); } /** * 2. ฟังก์ชันรับข้อมูลแบบ POST (จุดที่ขาดไป!) * ทำหน้าที่เป็นประตูรับข้อมูลจากหน้าเว็บ (fetch) มาประมวลผล */ function doPost(e) { try { // รับข้อมูล JSON ที่ส่งมาจากหน้าเว็บ var requestData = JSON.parse(e.postData.contents); var action = requestData.action; if (action === 'savePhoto') { var result = savePhotoToDriveAndSheet(requestData.image); return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // กรณีขอข้อมูลแกลลอรี่ if (action === 'getGallery') { var gallery = getGalleryData(); return ContentService.createTextOutput(JSON.stringify({ success: true, gallery: gallery })) .setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } /** * 3. ฟังก์ชันบันทึกรูปลง Drive และ Sheets */ function savePhotoToDriveAndSheet(base64Data) { try { var data = base64Data.split(',')[1]; var blob = Utilities.newBlob(Utilities.base64Decode(data), 'image/jpeg', 'NOKYOONG_' + new Date().getTime() + '.jpg'); // บันทึกลง Drive var file = DriveApp.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); // บันทึกลง Sheets var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var fileUrl = file.getUrl(); var timestamp = new Date(); // คอลัมน์ A: วันที่, คอลัมน์ B: ลิงก์, คอลัมน์ C: ID อ้างอิง sheet.appendRow([timestamp, fileUrl, file.getId()]); return { success: true, url: fileUrl, id: file.getId() }; } catch (error) { return { success: false, error: error.toString() }; } } /** * 4. ฟังก์ชันดึงข้อมูลรูปภาพล่าสุดมาแสดงในแกลลอรี่ */ function getGalleryData() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = sheet.getDataRange().getValues(); var gallery = []; // วนลูปดึงข้อมูลจากล่างขึ้นบน (รูปล่าสุดอยู่บน) for (var i = data.length - 1; i >= 1; i--) { if (data[i][1]) { // ถ้ามีลิงก์รูป gallery.push({ date: data[i][0], url: data[i][1], id: data[i][2] || i }); } if (gallery.length >= 20) break; // ดึงแค่ 20 รูปล่าสุดพอครับ } return gallery; }
